export default {
    data() {
        return {
            img: null,
            loaded: false
        }
    },
    props: ['src', 'className', 'id'],
    attached() {
        this.loadImg();
    },
    watch: {
        'src': function() {
            this.loadImg();
        }
    },
    methods: {
        loadImg() {
            this.loaded = false;
            if (this.src) {
                this.img = new window.Image();
                this.img.onload = () => {
                    this.loaded = true;
                };
                this.img.src = this.src;
            }
        }
    }
}
